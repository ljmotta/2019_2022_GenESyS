136 void Release::setQueueName(std::string _queueName) {
137  if (_queue->getName() != _queueName) {
138  Queue* queueWithNewName = (Queue*) _model->getInfraManager()->getInfrastructure(Util::TypeOf<Queue>(), _queueName);
139  if (queueWithNewName == nullptr) { // there is no queue with the new name
140  if (!_queue->isLinked()) { // no one else uses it. Only change the name
141  _queue->setName(_queueName);
142  } else { // it is linked. Create a new one
143  _queue = new Queue(_model);
144  _queue->setName(_queueName);
145  }
146  } else { // there is another queue with the same name
147  if (!_queue->isLinked()) { // no one else uses it. It can be removed
148  _model->getInfraManager()->getInfrastructures(Util::TypeOf<Queue>())->remove(_queue);
149  _queue->~Queue();
150  } else { // there is another one using the queue with old name. Let it there
151  _queue->removeLink();
152  }
153  _queue = queueWithNewName;
154  _queue->addLink();
155  }
156  }
157 }
