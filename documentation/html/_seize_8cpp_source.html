102 void Seize::setQueueName(std::string _queueName) {
103  if (_queue->getName() != _queueName) {
104  Queue* queueNewName = (Queue*) _model->getInfrastructure(Util::TypeOf<Queue>(), _queueName);
105  if (queueNewName == nullptr) { // there is no queue with the new name
106  if (!_queue->isLinked()) { // no one else uses it. Only change the name
107  _queue->setName(_queueName);
108  } else { // it is linked. Create a new one
109  _queue = new Queue();
110  _queue->setName(_queueName);
111  }
112  } else { // there is another queue with the same name
113  if (!_queue->isLinked()) { // no one else uses it. It can be removed
114  _model->getInfrastructures(Util::TypeOf<Queue>())->remove(_queue);
115  _queue->~Queue();
116  } else { // there is another one using the queue with old name. Let it there
117  _queue->removeLink();
118  }
119  _queue = queueNewName;
120  _queue->addLink();
121  }
122  }
123 }
